        fetchMovieDetails();
      }

        <p className="bold">Year: {movieDetail.Year}</p>
        <p className="bold">Genre: {movieDetail.Genre}</p>
        <p><strong>Review:</strong> {userReview}</p>
        <p><strong>Rating:</strong> {userRating}</p>
